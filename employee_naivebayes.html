library(caTools)
## Warning: package 'caTools' was built under R version 3.4.4

split = sample.split(empdata1, SplitRatio = 0.70)
traindata = subset(empdata1, split == TRUE)
testdata = subset(empdata1, split == FALSE)
library(klaR)

nb = NaiveBayes(Attrition ~ EnvironmentSatisfaction + YearsInCurrentRole  +  JobInvolvement + OverTime , data = traindata)
predictions = predict(nb, testdata)
library(caret)
